EXAMPLE-22 : Display Multi Dim Array in table format 
 object arraydisp {
   def main(args: Array[String]) = {
     val products = Array(Array("P01", 2.00), Array("P02", 4.00), Array("P03", 6.00))
     println("Product ID\tPrice")
     for (p <- products) {
       println(p(0) + "\t\t" + p(1))
     }
  }
}
The above program displays the contents of a multi dimensional array in a tabular format
EXAMPLE-23 : Product Price calculator ver 1 
 import scala.io.StdIn.readLine
object productprice {
   def main(args: Array[String]) = {

     //Input Parameters
     println("Please enter Product id P01 /P02 / P03:")
     val inputproduct = readLine()
     println("Please enter Quantity in numbers:")
     val inputqty = readLine()


     val qty: Double = inputqty.toDouble

     var unitprice = 0.00
     var totalprice = 0.00

     //product array - that has productid and Unit price combination
     val products = Array(Array("P01", 1.22), Array("P02", 4.00), Array("P03", 6.00))

     println("Product ID"+ "," + "Unit Price" + "," + "Qty" + "," + "Total Price")

     for (p <- products) {
       if (p(0) == inputproduct) {
       unitprice = p(1).asInstanceOf[Double]
       totalprice = qty * unitprice
       println(p(0) + "," + unitprice + "," + qty + "," + totalprice)
       }
     }
  }
}
The above program takes in the input Product id and quantity. The Unit cost of the product is obtained by reading multidimensional product array that has product id and Unit price combinations. The unit cost is multiplied to quantity to give the product price.
EXAMPLE-24 : Function Basics
 object functionbasics {
   def main(args: Array[String]) = {

     //Basic function
     def greeting(name: String, age: Int): String =
       "Hi , my name is " + name + " and I am " + age + " years old"

     println(greeting("John", 21))

     //Recursive function
     def factorial(n: Int): Int =
       if (n <= 1) 1
       else n * factorial(n - 1)

     println(factorial(7))

     //Recursive function -Prints the fibonacci number
     def fibonacci(n: Int): Int =
       if (n <= 2) 1
       else fibonacci(n - 1) + fibonacci(n - 2)

     println(fibonacci(7))

     //Recursive function Prime number test

     def primecheck(n: Int): Boolean = {

       def primeuntil(t: Int): Boolean =
         if (t <= 1) true
         else n % t != 0 && primeuntil(t - 1)

       primeuntil(n / 2)
     }
       println(primecheck(53))
       println(primecheck(2001))
       println(primecheck(605))

   }
}
The above example showcases basics of functions. Recursive and auxiliary.
